def self.connect_host(host, port, game, name, user, pass)
  $_api_socket.puts "CLIENT CONNECT?#{host}&#{port}&#{user}&#{pass}&#{game}&#{name}\n"
  $_api_socket.gets('\0').chomp('\0').to_s
end
